Example 2: Using JavaScript to Open the Widget

JavaScript Code:

// Method 1: With click handler
<button onclick="window.UserBird.open(this)">Open Feedback Widget</button>

// Method 2: Using JavaScript
document.getElementById('my-feedback-button').addEventListener('click', function() {
  window.UserBird.open(this);
});

Installation Instructions

Add this snippet to your HTML before the closing </body> tag:

<!-- Userbird Widget Initialization -->
<script>
  (function(w,d,s) {
    // Initialize Usermonk widget
    w.UserMonk = w.UserMonk || {};
    w.UserMonk.formId = "YOUR_FORM_ID";
    
    // Optional: Add user information
    w.UserMonk.user = {
      id: 'user-123',                 // Your user's ID
      email: 'user@example.com',      // User's email
      name: 'John Doe'                // User's name
    };
    
    // Load the widget script
    s = d.createElement('script');
    s.src = 'https://usermonk.netlify.app/widget.js';
    s.onload = function() {
      console.log('Usermonk widget loaded successfully');
    };
    s.onerror = function(error) {
      console.error('Failed to load Usermonk widget');
    };
    d.head.appendChild(s);
  })(window, document);
</script>
